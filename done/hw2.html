def normalize(x):
    s,m = np.std(x), np.mean(x)
    return [(a - m)/s for a in x]

def corr(x,y):
    return round((sum(map(o.mul, normalize(x), normalize(y))) / len(x)),3)

def remove_outliers(t, label):
    q1,q3 = np.percentile(t[label], 25.0), np.percentile(t[label], 75.0)
    max_val, min_val = q3 + 1.5 * (q3 - q1), q1 - 1.5 * (q3 - q1)
    return t.filter_rows(lambda row: min_val < row[label] < max_val)
